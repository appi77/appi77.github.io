мқёмҰқм„ң (self-signed) м Ғмҡ©

1лӢЁкі„: лЈЁнҠё CA нӮӨ л°Ҹ мқёмҰқм„ң мғқм„ұ
# 1. лЈЁнҠё CA к°ңмқё нӮӨ мғқм„ұ
openssl genrsa -out rootCA.key 4096
# 2. лЈЁнҠё CA мқёмҰқм„ң мғқм„ұ (мң нҡЁкё°к°„ 10л…„, мһҗмІҙ м„ңлӘ…)
openssl req -x509 -new -nodes -key rootCA.key -sha256 -days 3650 -out rootCA.crt \
-subj "/C=KR/ST=Seoul/L=Seoul/O=Penta System/OU=IT/CN=MyRootCA"

рҹ“ң 1лӢЁкі„: OpenSSL м„Өм • нҢҢмқј(optional)

#openssl.cnf
[ req ]
default_bits       = 2048
distinguished_name = req_distinguished_name
req_extensions     = req_ext
#x509_extensions    = v3_ca
prompt             = no

[ req_distinguished_name ]
C = KR
ST = Seoul
L = Seoul
O = Penta System
OU = IT
CN = dam-a.penta.co.kr

[ req_ext ]
subjectAltName = @alt_names

[ alt_names ]
DNS.1 = dam-a.penta.co.kr

рҹӣ пёҸ 2лӢЁкі„: м„ңлІ„ мқёмҰқм„ңмҡ© нӮӨ л°Ҹ CSR мғқм„ұ

# м„ңлІ„ нӮӨ мғқм„ұ
openssl genrsa -out tls.key 2048

# CSR мғқм„ұ (SANмқ„ нҸ¬н•Ён•ҳлҠ” м„Өм • нҢҢмқј мӮ¬мҡ©)
openssl req -new -key tls.key -out tls.csr -config openssl.cnf

рҹ”Ҹ 3лӢЁкі„: лЈЁнҠё CAлЎң м„ңлӘ… (мқёмҰқм„ң л°ңкёү)

openssl x509 -req -in tls.csr -CA rootCA.crt -CAkey rootCA.key -CAcreateserial \
-out tls.crt -days 365 -sha256 -extensions req_ext -extfile openssl.cnf

вң… 4. Secretмқ„ мӮ¬мҡ©н•ҳлҠ” Deployment/Ingress мһ¬мӢңмһ‘ (н•„мҡ” мӢң)

мқјл¶Җ Ingress ControllerлӮҳ DeploymentлҠ” Secret ліҖкІҪл§ҢмңјлЎң мһҗлҸҷ л°ҳмҳҒлҗҳм§Җ м•ҠмңјлҜҖлЎң, лӢӨмқҢ мӨ‘ н•ҳлӮҳлҘј н•ҳм„ёмҡ”:
