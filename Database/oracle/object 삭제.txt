-- 모든 테이블 삭제 스크립트 생성
SELECT 'DROP TABLE "' || table_name || '" CASCADE CONSTRAINTS;' FROM user_tables;
-- 모든 시퀀스 삭제 스크립트 생성
SELECT 'DROP SEQUENCE "' || sequence_name || '";' FROM user_sequences;
-- (나머지 오브젝트도 동일한 방식으로 추출 후 실행)

더 깔끔한 전체 삭제 순서

SELECT 'DROP TABLE "' || table_name || '" CASCADE CONSTRAINTS;' 
FROM user_tables 
WHERE iot_type IS NULL OR iot_type = 'IOT';

시퀀스: SELECT 'DROP SEQUENCE "' || sequence_name || '";' FROM user_sequences;

뷰: SELECT 'DROP VIEW "' || view_name || '";' FROM user_views;

프로시저/함수: SELECT 'DROP ' || object_type || ' "' || object_name || '";' FROM user_objects WHERE object_type IN ('PROCEDURE', 'FUNCTION', 'PACKAGE');
